API Interceptors

src/config/interceptors.js 
import { fetchBaseQuery } from '@reduxjs/toolkit/query/react';

export const baseQuery = fetchBaseQuery({
    baseUrl: 'http://localhost:3001',
    prepareHeaders: (headers, { getState }) => {
        const token = localStorage.getItem('token');
        if (token) {
            headers.set('authorization', `Bearer ${token}`);
        }
        headers.set('Content-Type', 'application/json');
        headers.set('Accept', 'application/json');
        return headers;
    },
});

export const responseHandler = async (response) => {
    if (response.error) {
        if (response.error.status === 401) {
            localStorage.removeItem('token');
            window.location.href = '/login';
        }
        return Promise.reject(response.error);
    }
    return response;
};

export const requestHandler = async (args, api, extraOptions) => {
    return args;
};

API Slice Implementation

src/store/api/apiSlice.js 
import { createApi } from '@reduxjs/toolkit/query/react';
import { API_ENDPOINTS } from '../../config/apiConfig';
import { baseQuery, responseHandler, requestHandler } from '../../config/interceptors';

export const apiSlice = createApi({
    reducerPath: 'api',
    baseQuery: async (args, api, extraOptions) => {
        const modifiedArgs = await requestHandler(args, api, extraOptions);
        const result = await baseQuery(modifiedArgs, api, extraOptions);
        return responseHandler(result);
    },
    endpoints: (builder) => ({
        sendOtp: builder.mutation({
            query: (mobileNumber) => ({
                url: API_ENDPOINTS.SEND_OTP,
                method: 'POST',
                body: { mobile_num: mobileNumber },
            }),
        }),
        verifyOtp: builder.mutation({
            query: ({ mobileNumber, otp }) => ({
                url: API_ENDPOINTS.VERIFY_OTP,
                method: 'POST',
                body: { 
                    mobile_num: mobileNumber,
                    otp: otp
                },
            }),
        }),
        registerUser: builder.mutation({
            query: (userData) => ({
                url: API_ENDPOINTS.REGISTER_USER,
                method: 'POST',
                body: userData,
            }),
        }),
        login: builder.mutation({
            query: (credentials) => ({
                url: API_ENDPOINTS.LOGIN,
                method: 'POST',
                body: credentials,
            }),
        }),
    }),
});

export const { 
    useSendOtpMutation,
    useVerifyOtpMutation,
    useRegisterUserMutation,
    useLoginMutation 
} = apiSlice;

Component Implementation Examples

OTP Verification Component

src/components/OtpVerification.jsx 
import React, { useState } from 'react';
import { useNavigate } from 'react-router-dom';
import { useVerifyOtpMutation } from '../store/api/apiSlice';
import { toast } from 'react-toastify';

const OtpVerification = () => {
    const [otp, setOtp] = useState(['', '', '', '', '', '']);
    const navigate = useNavigate();
    const [verifyOtp, { isLoading }] = useVerifyOtpMutation();

    const handleChange = (index, value) => {
        if (value.length <= 1 && /^\d*$/.test(value)) {
            const newOtp = [...otp];
            newOtp[index] = value;
            setOtp(newOtp);
        }
    };

    const handleSubmit = async (e) => {
        e.preventDefault();
        const otpValue = otp.join('');
        
        if (otpValue.length !== 6) {
            toast.error('Please enter a valid 6-digit OTP');
            return;
        }

        try {
            await verifyOtp({
                mobileNumber: localStorage.getItem('tempMobile'),
                otp: otpValue
            }).unwrap();
            
            toast.success('OTP verified successfully!');
            navigate('/registration');
        } catch (err) {
            toast.error(err.data?.message || 'OTP verification failed');
        }
    };

    return (
        

            {otp.map((digit, index) => (
                 handleChange(index, e.target.value)}
                    maxLength={1}
                />
            ))}
            
        

    );
};

Login Component

import React, { useState } from 'react';
import { useNavigate } from 'react-router-dom';
import { useLoginMutation } from '../store/api/apiSlice';
import { toast } from 'react-toastify';

const Login = () => {
    const navigate = useNavigate();
    const [login, { isLoading }] = useLoginMutation();
    const [formData, setFormData] = useState({
        email: '',
        password: '',
    });

    const handleChange = (e) => {
        const { name, value } = e.target;
        setFormData(prev => ({
            ...prev,
            [name]: value
        }));
    };

    const handleSubmit = async (e) => {
        e.preventDefault();
        
        try {
            const response = await login(formData).unwrap();
            localStorage.setItem('token', response.data.token);
            localStorage.setItem('user', JSON.stringify(response.data.user));
            
            toast.success('Login successful!');
            navigate('/dashboard');
        } catch (err) {
            toast.error(err.data?.message || 'Login failed');
        }
    };

    return (
        

            
            
            
        

    );
};

Best Practices

Important Guidelines

  • Always handle loading states using the isLoading property from mutations
  • Implement proper error handling with try-catch blocks
  • Use token management in interceptors for authentication
  • Provide user feedback using toast notifications
  • Validate forms before making API calls
  • Keep components focused and maintainable
  • Use proper TypeScript types when applicable
  • Follow consistent naming conventions
